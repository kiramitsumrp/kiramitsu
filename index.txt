const { Client, GatewayIntentBits, Events } = require('discord.js');

// 1. Inisialisasi Client
const client = new Client({
    intents: [
        GatewayIntentBits.Guilds,
        GatewayIntentBits.GuildMessages,
        GatewayIntentBits.MessageContent, // Pastikan 'Message Content Intent' sudah NYALA di Developer Portal
    ],
});

// 2. Event saat bot siap (Menggunakan 'ClientReady' sesuai saran warning tadi)
client.once(Events.ClientReady, (readyClient) => {
    console.log(`Siap! Login sebagai ${readyClient.user.tag}`);
});

// 3. Event saat ada pesan masuk
client.on(Events.MessageCreate, (message) => {
    // Abaikan jika yang kirim pesan adalah bot itu sendiri
    if (message.author.bot) return;

    // Test respon sederhana
    if (message.content.toLowerCase() === 'halo') {
        message.reply('Halo juga! Ada yang bisa saya bantu di Kiramitsu?');
    }
    
    if (message.content.toLowerCase() === 'ping') {
        message.reply('Pong! 🏓');
    }
});
